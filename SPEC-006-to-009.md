# SPEC-006 — Oversee Application Documents
## Maps to: UC-03-01

## Goal
Student Affairs Officer reviews submitted applications, previews documents in-browser (PDF viewer, no download required), and takes one of three actions: Approve Verification → UNDER_REVIEW, Request Correction → CORRECTION_REQUESTED, Reject → REJECTED. Every action logged with timestamp + officer ID (SRS SR2).

SPEC-004

```python
def convert_gpa_yok(gpa_4: float) -> float:
    """
    Official YÖK table-based conversion (NOT a linear formula).
    Must be deterministic: same input → same output always.
    Ranking formula from SRS UC-04-03:
      Exam Component = (yks_score / base_score) × 100 × 0.90
      GPA Component  = converted_gpa_100 × 0.10
      Transfer Score = Exam Component + GPA Component
    """
    YOK_TABLE = {
        4.00: 100.00, 3.75: 95.83, 3.50: 88.33, 3.25: 80.83,
        3.00: 76.67, 2.75: 72.50, 2.50: 65.00, 2.25: 57.50,
        2.00: 50.00, 1.75: 42.50, 1.50: 35.00, 1.25: 27.50,
        1.00: 20.00,
    }
    # Use nearest table entry; for values between entries interpolate linearly
```

```python
class EligibilityEngine:
    async def evaluate_department_conditions(
        self, application_id: UUID, evaluator_id: UUID, notes: str | None
    ) -> DepartmentEvaluation
    # 1. Load DepartmentRequirement records for program
    # 2. Evaluate each rule against academic_record:
    #    MIN_GPA    → gpa_4 >= rule_value
    #    MIN_YKS    → yks_score >= rule_value
    #    MIN_CREDITS → credits_completed >= rule_value
    #    REQUIRED_DOC → doc of type exists and status=ACCEPTED
    #    CORE_COURSE_GRADE → check course grade in transcript (UC-04-02 manual mapping)
    # 3. Store each result in eligibility_checks with "Met"/"Not Met" (SRS UC-04-02 step 5)
    # 4. Auto-check completes ≤ 2 seconds (SRS perf)
    # 5. If any rule fails: DepartmentEvaluation(passed=False), notify
    # 6. AuditLog action="DEPT_CONDITIONS_EVALUATED"

    async def manual_course_mapping(
        self, application_id: UUID, external_course: str,
        rule_key: str, evaluator_id: UUID
    ) -> EligibilityCheck
    # Handles UC-04-02.AC-01 (manual course mapping)
    # Logs manual intervention for audit purposes (SRS SR2)
```
